            function checkCriterionIsMet (key, criteriaMet) {
                var keys = key.split(",");
                var result, index;
 
                // Check for logic key values
                // | = (any criterion in list will do)
                // & = (all criteria in list must be met)
                if ("|" === keys[0]) {
                    keys.shift();
                    result = [];
                    for (index=0; index < keys.length; ++index) {
                        if (-1 !== criteriaMet.indexOf(keys[index])) {
                            result.push(true);
                        } else {
                            result.push(false);
                        }
                    }
                    return result;
                } else if ("&" === keys[0]) {
                    keys.shift();
                    result = true;
                    for (index=0; index < keys.length; ++index) {
                        if (-1 === criteriaMet.indexOf(keys[index]))
                            result = false;
                    }
                    return [result];
                }
                if (keys.length > 1) {
                    $log.error("Multiple keys without a logic key value! (" + keys[0] + ") " + key + " -> " + keys);
                }
                return [(-1 !== criteriaMet.indexOf(keys[0]))];
            }
